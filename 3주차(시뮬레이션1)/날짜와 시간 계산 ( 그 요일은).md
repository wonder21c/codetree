```c++
#include <iostream>
#include <string>
using namespace std;

int main() {
    int m1, m2, d1, d2;
    string doth;
    cin >> m1 >> d1 >> m2 >> d2 >> doth;

    int all_days = 1;
    string days_of_the_week[7] = {"Mon", "Tue", "Wed", "Thu", "Fri", "Sat", "Sun"};
    int num_of_days[13] = {0, 31, 29, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};

    while (true) {
        if (m1 == m2 && d1 == d2) {
            break;
        }

        d1++;
        all_days++;
        if (d1 > num_of_days[m1]) {
            m1++;
            d1 = 1;
        }


    }
 

    int days_of_week_counts[7] = {0};

    for (int i = 0; i < all_days; i++) {
        days_of_week_counts[i % 7]++;
    }

    for (int i = 0; i < 7; i++) {
        if (days_of_the_week[i] == doth) {
            cout << days_of_week_counts[i];
        }
    }

    return 0;
}
```
